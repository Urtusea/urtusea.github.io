# [Luogu P1083 - [NOIP2012 提高组] 借教室](https://www.luogu.com.cn/problem/P1083)

## Tag: 二分，差分

## 题意

问你最早的不满足订单编号

## 思路

二分 + 差分模拟即可

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

auto code() -> void {
    int n, m;
    std::cin >> n >> m;

    std::vector<Int> a(n + 1);
    for (int i = 1; i <= n; i++)
        std::cin >> a[i];
    
    std::vector<std::array<Int, 3>> opt(m + 1);
    for (int i = 1; i <= m; i++)
        std::cin >> opt[i][0] >> opt[i][1] >> opt[i][2];
    
    auto check = [&](Int x) -> bool {
        std::vector<Int> d(n + 2);
        for (int i = 1; i <= x; i++) {
            d[opt[i][1]] += opt[i][0];
            d[opt[i][2] + 1] -= opt[i][0];
        }
        for (int i = 1; i <= n; i++) {
            d[i] += d[i - 1];
            if (d[i] > a[i]) return true;
        }
        return false;
    };

    Int l = 1, r = m;
    while (l + 1 < r) {
        Int mid = l + r >> 1;
        if (check(mid))
            r = mid;
        else
            l = mid;
    }

    if (check(l)) {
        std::cout << -1 << '\n';
        std::cout << l << '\n';
    } else if (check(r)) {
        std::cout << -1 << '\n';
        std::cout << r << '\n';
    } else {
        std::cout << 0 << '\n';
    }
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

# [Luogu P1010 - [NOIP1998 普及组] 幂次方](https://www.luogu.com.cn/problem/P1010)

## Tag: 递归

## 题意

将所给的数 $n$ 依据题目条件进行分解

## 思路

递归跑一遍即可 

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

auto code() -> void {
    int n;
    std::cin >> n;

    std::string ans;
    auto solve = [&](auto &&self, int x) -> void {
        for (int i = std::__lg(x); i >= 0; i--) {
            if (x >> i & 1) {
                if (i == 0) {
                    ans += "2(0)";
                } else if (i == 1) {
                    ans += "2";
                } else {
                    ans += "2(";
                    self(self, i);
                    ans += ")";
                }
                x ^= 1 << i;
                if (x != 0)
                    ans += "+";
            }
        }
    };

    solve(solve, n);
    std::cout << ans << '\n';
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

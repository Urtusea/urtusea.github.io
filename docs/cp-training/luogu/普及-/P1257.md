# [Luogu P1257 - 平面上的最接近点对](https://www.luogu.com.cn/problem/P1257)

## Tag: 枚举

## 题意

查找二维平面上的 $n$ 个点之间的最小距离

## 思路

暴力枚举

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

#define D(X, Y) ((X.x - Y.x) * (X.x - Y.x) + (X.y - Y.y) * (X.y - Y.y))

constexpr Int INF = 3e18;

struct pos {
    Int x, y;
    
    pos()
    : x(0), y(0) {}
    pos(Int _x, Int _y)
    : x(_x), y(_y) {}
    
    auto input() -> void {
        std::cin >> x >> y;
    }
};

auto code() -> void {
    int n;
    std::cin >> n;

    std::vector<pos> p(n + 1);
    for (int i = 1; i <= n; i++)
        p[i].input();
    
    Int ans = INF;
    for (int i = 1; i <= n; i++) {
        for (int j = i + 1; j <= n; j++) {
            ans = std::min(ans, D(p[i], p[j]));
        }
    }

    std::cout << sqrt((double)(ans)) << '\n';
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    std::cout << std::fixed << std::setprecision(4);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

# [Luogu P3722 - [AH2017/HNOI2017] 影魔](https://www.luogu.com.cn/problem/P3722)

## Tag: 线段树，扫描线

## 题意

给你一个大小为 $n$ 的排列，共询问 $m$ 个区间 $[L, R]$，如果存在 $L \le i < j \le R$，$L$ 和 $R$ 是最大值和次大值获得 $X$ 点贡献，如果只有一个是最大值，无次大值，获得 $Y$ 点贡献，问区间 $[L, R]$ 的贡献是多少

## 思路

考虑将贡献转化，转换成离线二维数点，将每个区间看作 $(l, r)$，那么一次查询 $[L, R]$ 则是查询 $(L, L)$ 到 $(R, R)$ 这个矩形内有多少个满足 $X$ 和 $Y$ 贡献的点

预处理前一个的最大值和后一个的最大值，分别为 $Prev$ 和 $Next$，那么有:

- $(Prev_i, Next_i)$ 有贡献 $X$
- $([Prev_i + 1, i - 1], Next_i)$ 有贡献 $Y$
- $(Prev_i, [i + 1, Next_i - 1])$ 有贡献 $Y$

接下来二维数点板子即可

#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

template <typename Info, typename Lazy> struct segment_tree {
public:
    #define lson (p << 1)
    #define rson (p << 1 | 1)
    segment_tree(int _n)
    : n(_n), f(4 << std::__lg(_n)), g(4 << std::__lg(_n)) {}
    segment_tree(std::vector<Info> info)
    : segment_tree(info.size() - 1) {
        auto build = [&](auto &&self, int p, int l, int r) -> void {
            if (l == r) return (void)(f[p] = info[l]);
            int m = l + r >> 1;
            self(self, lson, l, m);
            self(self, rson, m + 1, r);
            push_up(p);
        };
        build(build, 1, 1, n);
    }
    auto push_up(int p) -> void {
        f[p] = f[lson] + f[rson];
    }
    auto push_down(int p) -> void {
        f[lson] += g[p], g[lson] += g[p];
        f[rson] += g[p], g[rson] += g[p];
        g[p] = Lazy();
    }
    auto update(int p, int l, int r, int L, int R, Lazy x) -> void {
        if (R <  l || r <  L) return;
        if (L <= l && r <= R) return (void)(f[p] += x, g[p] += x);
        push_down(p);
        int m = l + r >> 1;
        update(lson, l, m, L, R, x);
        update(rson, m + 1, r, L, R, x);
        push_up(p);
    }
    auto query(int p, int l, int r, int L, int R) -> Info {
        if (R <  l || r <  L) return Info();
        if (L <= l && r <= R) return f[p];
        push_down(p);
        int m = l + r >> 1;
        return query(lson, l, m, L, R) + query(rson, m + 1, r, L, R);
    }
    auto find_first(int p, int l, int r, int L, int R, auto &check) -> int {
        if (R < l || r > L || !check(f[p])) return -1;
        if (l == r) return l;
        int m = l + r >> 1;
        int res = find_first(lson, l, m, L, R, check);
        if (res == -1)
            return find_first(rson, m + 1, r, L, R, check);
        else
            return res;
    }
    auto find_last(int p, int l, int r, int L, int R, auto &check) -> int {
        if (R < l || r > L || !check(f[p])) return -1;
        if (l == r) return l;
        int m = l + r >> 1;
        int res = find_last(rson, m + 1, r, L, R, check);
        if (res == -1)
            return find_last(lson, l, m, L, R, check);
        else
            return res;
    }
    #undef lson
    #undef rson
private:
    int n;
    std::vector<Info> f;
    std::vector<Lazy> g;
};

auto get_prev(int n, std::vector<int> &a) -> std::vector<int> {
    static std::vector<int> prev, stk;
    prev.assign(n + 1, 0);
    stk.clear();

    for (int i = 1; i <= n; i++) {
        while (!stk.empty() && a[stk.back()] < a[i])
            stk.pop_back();
        if (!stk.empty())
            prev[i] = stk.back();
        stk.push_back(i);
    }

    return prev;
}

auto get_next(int n, std::vector<int> &a) -> std::vector<int> {
    static std::vector<int> next, stk;
    next.assign(n + 2, n + 1);
    stk.clear();

    for (int i = n; i >= 1; i--) {
        while (!stk.empty() && a[stk.back()] < a[i])
            stk.pop_back();
        if (!stk.empty())
            next[i] = stk.back();
        stk.push_back(i);
    }

    return next;
}

struct lazy {
    Int x, y;
    lazy()
    : x(0), y(0) {}
    lazy(Int _x, Int _y)
    : x(_x), y(_y) {}

    lazy &operator += (lazy tag) {
        x += tag.x;
        y += tag.y;
        return *this;
    }
};

struct info {
    Int x, y, siz;
    info()
    : x(0), y(0), siz(0) {}
    info(Int _x, Int _y, Int _siz)
    : x(_x), y(_y), siz(_siz) {}

    info &operator += (lazy tag) {
        x += tag.x * siz;
        y += tag.y * siz;
        return *this;
    }

    friend info operator + (info l, info r) {
        return info(l.x + r.x, l.y + r.y, l.siz + r.siz);
    }
};

auto code() -> void {
    int n, m, x, y;
    std::cin >> n >> m >> x >> y;

    std::vector<int> a(n + 1);
    for (int i = 1; i <= n; i++)
        std::cin >> a[i];
    
    std::vector<int> prev = get_prev(n, a);
    std::vector<int> next = get_next(n, a);

    std::vector<std::array<int, 5>> opt;
    for (int i = 1; i <= m; i++) {
        int l, r;
        std::cin >> l >> r;
        opt.push_back({l - 1, l, r, i, -1});
        opt.push_back({r, l, r, i,  1});
    }
    std::sort(opt.begin(), opt.end());

    std::vector<std::array<int, 4>> upd;
    for (int i = 1; i <= n; i++) {
        if (1 <= prev[i] && next[i] <= n)
            upd.push_back({next[i], prev[i], prev[i], 1});
        if (1 <= prev[i] && next[i] - 1 > i)
            upd.push_back({prev[i], i + 1, next[i] - 1, 0});
        if (n >= next[i] && prev[i] + 1 < i)
            upd.push_back({next[i], prev[i] + 1, i - 1, 0});
    }
    std::sort(upd.begin(), upd.end());

    int idx = 0;
    std::vector<Int> res(m + 1);
    segment_tree<info, lazy> seg(std::vector<info>(n + 1, info(0, 0, 1)));
    for (auto &[now, l, r, id, tag] : opt) {
        while (idx < upd.size() && upd[idx][0] <= now) {
            auto [u, l, r, v] = upd[idx++];
            seg.update(1, 1, n, l, r, lazy(v, v ^ 1));
        }
        auto [a, b, c] = seg.query(1, 1, n, l, r);
        res[id] += (a * x + b * y) * tag;
        if (tag == 1) res[id] += (r - l) * x;
    }

    for (int i = 1; i <= m; i++)
        std::cout << res[i] << '\n';
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}

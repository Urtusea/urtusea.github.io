# [Luogu P4198 - 楼房重建](https://www.luogu.com.cn/problem/P4198)

## Tag: 线段树，二分

## 题意

给你一个大小为 $n$ 的序列，共有 $m$ 次操作，每次操作后查找从 $1$ 开始的最长的递增序列，其中忽略为 $0$ 的项

## 思路

考虑最长递增子序列是如何维护的，二分新加进来的数所能放置的位置

如果是两个递增序列合并呢？

设左递增序列为 $p_1, p_2, ..., p_x$，右递增序列为 $q_1, q_2, ..., q_y$

那么合并两序列需要找到 $p_x$ 在 $q_1, ..., q_y$ 中所能处的位置

而我线段树上，已经将 $q_1, .., q_y$ 的信息存储了，故而只需要在线段树上二分即可

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

template <typename Info, typename Node> struct segment_tree {
public:
    #define lson (p << 1)
    #define rson (p << 1 | 1)
    segment_tree(int _n)
    : n(_n), f(4 << std::__lg(_n)) {}
    segment_tree(std::vector<Info> info)
    : segment_tree(info.size() - 1) {
        auto build = [&](auto &&self, int p, int l, int r) -> void {
            if (l == r) return (void)(f[p] = info[l]);
            int m = l + r >> 1;
            self(self, lson, l, m);
            self(self, rson, m + 1, r);
            push_up(p, l, r);
        };
        build(build, 1, 1, n);
    }
    auto maintain(int p, int l, int r, Node pre) -> int {
        if (f[p].pos <= pre)
            return 0;
        if (l == r)
            return pre < f[p].pos;
        int m = l + r >> 1;
        if (f[lson].pos <= pre)
            return maintain(rson, m + 1, r, pre);
        else
            return maintain(lson, l, m, pre) + f[p].val - f[lson].val;
    }
    auto push_up(int p, int l, int r) -> void {
        f[p].pos = std::max(f[lson].pos, f[rson].pos);
        f[p].val = f[lson].val + maintain(rson, (l + r >> 1) + 1, r, f[lson].pos);
    }
    auto update(int p, int l, int r, int u, Info x) -> void {
        if (u < l || r < u) return;
        if (l == r) return (void)(f[p] = x);
        int m = l + r >> 1;
        update(lson, l, m, u, x);
        update(rson, m + 1, r, u, x);
        push_up(p, l, r);
    }
    auto query(int p, int l, int r, int L, int R) -> int {
        if (R <  l || r <  L) return 0;
        if (L <= l && r <= R) return f[p].val;
        int m = l + r >> 1;
        return query(lson, l, m, L, R) + query(rson, m + 1, r, L, R);
    }
    #undef lson
    #undef rson
private:
    int n;
    std::vector<Info> f;
};

struct node {
    int x, y;
    node()
    : x(0), y(0) {}
    node(int _x, int _y)
    : x(_x), y(_y) {}

    friend bool operator < (node l, node r) {
        return (Int)l.y * r.x <  (Int)r.y * l.x;
    }
    friend bool operator <= (node l, node r) {
        return (Int)l.y * r.x <= (Int)r.y * l.x;
    }
};

struct info {
    node pos;
    int  val;
    info()
    : pos(node()), val(0) {}
    info(node _pos, int _val)
    : pos(_pos), val(_val) {}
};

auto code() -> void {
    int n, m;
    std::cin >> n >> m;

    std::vector<info> init(n + 1);
    for (int i = 1; i <= n; i++)
        init[i].pos.x = i;

    segment_tree<info, node> seg(init);
    for (int i = 1; i <= m; i++) {
        node tmp;
        std::cin >> tmp.x >> tmp.y;
        seg.update(1, 1, n, tmp.x, info(tmp, 1));
        std::cout << seg.query(1, 1, n, 1, n) << '\n';
    }
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

# [Luogu P2471 - [SCOI2007] 降雨量](https://www.luogu.com.cn/problem/P2471)

## Tag: 模拟，二分

## 题意

每次询问 $[L, R]$ 区间的端点值是否是区间最大值和次大值，且 $val_L > val_R$，中间有一些值没有给出

## 思路

处理每个数的前一个最大值位置，和后一个最大值位置，分类讨论即可

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

auto code() -> void {
    int n;
    std::cin >> n;

    std::map<int, int> cnt;
    std::vector<int> all = {(int)(-2e9)};
    std::vector<std::pair<int, int>> info(n + 1);
    for (int i = 1; i <= n; i++) {
        std::cin >> info[i].first >> info[i].second;
        cnt[info[i].first] = info[i].second;
        all.push_back(info[i].first);
    }
    info[0].first = all[0];

    std::vector<int> pre_max(n + 1), stk1;
    for (int i = 1; i <= n; i++) {
        while (!stk1.empty() && info[stk1.back()].second < info[i].second)
            stk1.pop_back();
        if (!stk1.empty())
            pre_max[i] = stk1.back();
        stk1.push_back(i);
    }

    std::vector<int> nxt_max(n + 1), stk2;
    for (int i = n; i >= 1; i--) {
        while (!stk2.empty() && info[stk2.back()].second < info[i].second)
            stk2.pop_back();
        if (!stk2.empty())
            nxt_max[i] = stk2.back();
        stk2.push_back(i);
    }

    int m;
    std::cin >> m;

    auto count = [&](int x) -> bool {
        return std::binary_search(all.begin(), all.end(), x);
    };

    auto rank = [&](int x) -> int {
        return std::lower_bound(all.begin(), all.end(), x) - all.begin();
    };

    for (int i = 1; i <= m; i++) {
        int x, y;
        std::cin >> x >> y;

        if (!count(x) && !count(y)) {
            std::cout << "maybe\n";
        } else if (!count(x)) {
            int idx_y = rank(y);
            if (x < info[pre_max[idx_y]].first) {
                std::cout << "false\n";
            } else {
                std::cout << "maybe\n";
            }
        } else if (!count(y)) {
            int idx_x = rank(x);
            if (nxt_max[idx_x] != 0 && info[nxt_max[idx_x]].first < y) {
                std::cout << "false\n";
            } else {
                std::cout << "maybe\n";
            }
        } else {
            int idx_x = rank(x), idx_y = rank(y);
            if (info[idx_x].second < info[idx_y].second) {
                std::cout << "false\n";
            } else if (idx_x < pre_max[idx_y]) {
                std::cout << "false\n";
            } else if (idx_y - idx_x != y - x) {
                std::cout << "maybe\n";
            } else {
                std::cout << "true\n";
            }
        }
    }
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

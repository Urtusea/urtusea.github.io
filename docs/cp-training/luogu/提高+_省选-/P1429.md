# [Luogu P1429 - 平面最近点对（加强版）](https://www.luogu.com.cn/problem/P1429)

## Tag: 分治，归并

## 题意

查找二维平面上的 $n$ 个点之间的最小距离

## 思路

考虑分治，每次将当前 $x$ 的中位数作为分界线，由于我的左部分最小 $dis$ 和我的右部分最小 $dis$ 是可以确定的，那么如果中间部分有更优的 $dis$ 的话则需要枚举

由于我令 $x$ 的中位数作为了分界线，那么也就有，$x - x_{mid} \le min(dis_l, dis_r)$ 的点才符合我去枚举的条件

然后再顺序枚举一下即可

总复杂度 $O(nlogn)$

## 代码

``` cpp
#include <bits/stdc++.h>

using Int  = int64_t;
using uInt = uint64_t;

#define P(X, Y) ((X - Y) * (X - Y))
#define D(X, Y) (P(X.x, Y.x) + P(X.y, Y.y))

constexpr Int INF = 4e18;

struct pos {
    Int x, y;
};

auto code() -> void {
    int n;
    std::cin >> n;

    std::vector<pos> p(n + 1);
    for (int i = 1; i <= n; i++)
        std::cin >> p[i].x >> p[i].y;
    std::sort(p.begin() + 1, p.end(), [](pos &l, pos &r) {
        return l.x < r.x;
    });

    auto merge = [&](int l, int m, int r) -> void {
        static std::vector<pos> tmp(n + 1);
        int idx = l, L = l, R = m + 1;
        while (L <= m && R <= r) {
            if (p[L].y < p[R].y)
                tmp[idx++] = p[L++];
            else
                tmp[idx++] = p[R++];
        }
        while (L <= m) tmp[idx++] = p[L++];
        while (R <= r) tmp[idx++] = p[R++];
        for (int i = l; i <= r; i++) p[i] = tmp[i];
    };

    auto dfs = [&](auto &&self, int l, int r) -> Int {
        if (l == r) return INF;
        if (l + 1 == r) {
            if (p[l].y > p[r].y)
                std::swap(p[l], p[r]);
            return D(p[l], p[r]);
        }

        int m = l + r >> 1;
        Int mid = p[m].x;
        Int dis = std::min(self(self, l, m), self(self, m + 1, r));
        merge(l, m, r);

        std::vector<pos> tmp;
        for (int i = l; i <= r; i++) {
            if (P(p[i].x, mid) < dis) tmp.push_back(p[i]);
        }

        for (int i = 0; i < tmp.size(); i++) {
            for (int j = i + 1; j < tmp.size() && P(tmp[i].y, tmp[j].y) < dis; j++) {
                dis = std::min(dis, D(tmp[i], tmp[j]));
            }
        }

        return dis;
    };

    std::cout << sqrt((double)(dfs(dfs, 1, n))) << '\n';
}

auto main(int argc, char *argv[], char *envp[]) -> int {
    std::cin.tie(nullptr)->sync_with_stdio(false);
    std::cout << std::fixed << std::setprecision(4);

    // for (int i = 1, n = (std::cin >> n, n); i <= n; ++i)
        code();

    return 0;
}
```

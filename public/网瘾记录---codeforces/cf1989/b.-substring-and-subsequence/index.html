    B. Substring and Subsequence

    前言： #

    • Links: B. Substring and Subsequence
    • Codeforces: 待更新
    • Clist: \(1142\)
    • Tags: 最长递增子序列，双指针
    • 时间复杂度: \(O(|a|·|b|)\)

    题意： #

    对于一个字符串 \(s\) ，给你其一子串 \(a\) 和一子序列 \(b\) ，询问你满足条件的 \(|s|\) 的最小值

    思路： #

    考虑 DP，令 \(f(i)\) 为 以 \(b_i\) 开头，与 \(a\) 匹配的最长公共子序列，不难得到答案为： \(|s| = a + b - min(f(i))\)

    代码： # 

    #include <bits/stdc++.h>
using namespace std;

void code(/* Copyright by Urtusea */)
{
    string a, b;
    cin >> a >> b;

    int n = a.size(), m = b.size();
    
    // 将字符串变为 "#a"，使得其下标从 1 开始，个人习惯罢了
    a = '#' + a;
    b = '#' + b;

    vector f(m + 1, 0LL);
    for (int i = 1; i <= m; i++)
    for (int j = 1; j <= n; j++)
    {
        if (f[i] + i > m) break;
        if (a[j] == b[f[i] + i]) f[i]++;
    }

    cout << n + m - *max_element(f.begin(), f.end()) << '\n';
}

int main(int argc, char *argv[], char *envp[])
{
    cin.tie(nullptr)->sync_with_stdio(false);

#ifdef Urtusea
    freopen("main.in",  "r", stdin );
    freopen("main.out", "w", stdout);
#endif

    for (int i = 1, n = (cin >> n, n); i <= n; i++)
    {
        code();
    }

    return 0;
}
